## 3월 1주차 문제
### 1번
```python
num = int(input())
strlist = []
result = []
for i in range(num):
    strlist.append(input())
    temp = list(strlist[i])
    retemp = []
    for t in temp:
        if (t == 'a') or (t == 'A'):
            retemp.append(t)
        elif (t == 'e') or (t == 'E'):
            retemp.append(t)
        elif (t == 'o') or (t == 'O'):
            retemp.append(t)
        elif (t == 'u') or (t == 'U'):
            retemp.append(t)
        elif (t == 'i') or (t == 'I'):
            retemp.append(t)
    if retemp:
        result.append(''.join(retemp))
    else:
        result.append("???")
        
for i in range(num):
    print(result[i])
```

### 2번
```python
num1, num2 = map(int, input().split())
divilist = []
temp = []
result = 2
for num in range(num1, num2 + 1):
    for i in range(2, num + 1):
        if num % i == 0:
            if i not in divilist:
                divilist.append(i)
            temp.append(i)

divilist.sort(reverse=True)
for i in divilist:
    if divilist.count(i) > divilist.count(result):
        result = i

print(result)
# 현재 이방법은 큰수일때 시간초과가 난다. 다르 방법을 생각하자!
``` 
